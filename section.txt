<script>
document.addEventListener("DOMContentLoaded", function () {
  const section = document.querySelector('.clinical-proven-section'); // Replace with your actual section class or ID
  const counters = document.querySelectorAll('.stat-percentage');

  const options = {
    threshold: .5 // Section must be 100% visible
  };

  const animateCount = (el) => {
    const target = parseInt(el.getAttribute('data-target'), 10);
    let current = 0;
    const duration = 1000;
    const stepTime = Math.max(10, duration / target);

    const step = () => {
      current++;
      el.textContent = current + '%';
      if (current < target) {
        setTimeout(step, stepTime);
      }
    };

    step();
  };

  const observer = new IntersectionObserver((entries, obs) => {
    entries.forEach(entry => {
      if (entry.isIntersecting) {
        // Animate all counters only once
        counters.forEach(el => {
          if (!el.classList.contains('counted')) {
            el.classList.add('counted');
            animateCount(el);
          }
        });
        obs.unobserve(entry.target); // Unobserve after first trigger
      }
    });
  }, options);

  if (section) {
    observer.observe(section);
  }
});
</script>
